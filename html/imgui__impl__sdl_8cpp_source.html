108 // You can read the io.WantCaptureMouse, io.WantCaptureKeyboard flags to tell if dear imgui wants to use your inputs.
109 // - When io.WantCaptureMouse is true, do not dispatch mouse input data to your main application.
110 // - When io.WantCaptureKeyboard is true, do not dispatch keyboard input data to your main application.
111 // Generally you may always pass all inputs to dear imgui, and hide them from your application based on those two flags.
112 // If you have multiple SDL events and some of them are not meant to be used by dear imgui, you may need to filter events based on their windowID field.
113 bool ImGui_ImplSDL2_ProcessEvent(const SDL_Event* event)
114 {
115  ImGuiIO& io = ImGui::GetIO();
116  ImGui_ImplSDL2_Data* bd = ImGui_ImplSDL2_GetBackendData();
117 
118  switch (event->type)
119  {
120  case SDL_MOUSEWHEEL:
121  {
122  if (event->wheel.x > 0) io.MouseWheelH += 1;
123  if (event->wheel.x < 0) io.MouseWheelH -= 1;
124  if (event->wheel.y > 0) io.MouseWheel += 1;
125  if (event->wheel.y < 0) io.MouseWheel -= 1;
126  return true;
127  }
128  case SDL_MOUSEBUTTONDOWN:
129  {
130  if (event->button.button == SDL_BUTTON_LEFT) { bd->MousePressed[0] = true; }
131  if (event->button.button == SDL_BUTTON_RIGHT) { bd->MousePressed[1] = true; }
132  if (event->button.button == SDL_BUTTON_MIDDLE) { bd->MousePressed[2] = true; }
133  return true;
134  }
135  case SDL_TEXTINPUT:
136  {
137  io.AddInputCharactersUTF8(event->text.text);
138  return true;
139  }
140  case SDL_KEYDOWN:
141  case SDL_KEYUP:
142  {
143  int key = event->key.keysym.scancode;
144  IM_ASSERT(key >= 0 && key < IM_ARRAYSIZE(io.KeysDown));
145  io.KeysDown[key] = (event->type == SDL_KEYDOWN);
146  io.KeyShift = ((SDL_GetModState() & KMOD_SHIFT) != 0);
147  io.KeyCtrl = ((SDL_GetModState() & KMOD_CTRL) != 0);
148  io.KeyAlt = ((SDL_GetModState() & KMOD_ALT) != 0);
149 #ifdef _WIN32
150  io.KeySuper = false;
151 #else
152  io.KeySuper = ((SDL_GetModState() & KMOD_GUI) != 0);
153 #endif
154  return true;
155  }
156  case SDL_WINDOWEVENT:
157  {
158  if (event->window.event == SDL_WINDOWEVENT_FOCUS_GAINED)
159  io.AddFocusEvent(true);
160  else if (event->window.event == SDL_WINDOWEVENT_FOCUS_LOST)
161  io.AddFocusEvent(false);
162  return true;
163  }
164  }
165  return false;
166 }

173  // Check and store if we are on a SDL backend that supports global mouse position
174  // ("wayland" and "rpi" don't support it, but we chose to use a white-list instead of a black-list)
175  bool mouse_can_use_global_state = false;
176 #if SDL_HAS_CAPTURE_AND_GLOBAL_MOUSE
177  const char* sdl_backend = SDL_GetCurrentVideoDriver();
178  const char* global_mouse_whitelist[] = { "windows", "cocoa", "x11", "DIVE", "VMAN" };
179  for (int n = 0; n < IM_ARRAYSIZE(global_mouse_whitelist); n++)

181  mouse_can_use_global_state = true;
182 #endif
183 
184  // Setup backend capabilities flags
185  ImGui_ImplSDL2_Data* bd = IM_NEW(ImGui_ImplSDL2_Data)();

296 static void ImGui_ImplSDL2_UpdateMousePosAndButtons()
297 {
298  ImGui_ImplSDL2_Data* bd = ImGui_ImplSDL2_GetBackendData();
299  ImGuiIO& io = ImGui::GetIO();
300 
301  ImVec2 mouse_pos_prev = io.MousePos;
302  io.MousePos = ImVec2(-FLT_MAX, -FLT_MAX);
303 
304  // Update mouse buttons
305  int mouse_x_local, mouse_y_local;
306  Uint32 mouse_buttons = SDL_GetMouseState(&mouse_x_local, &mouse_y_local);
307  io.MouseDown[0] = bd->MousePressed[0] || (mouse_buttons & SDL_BUTTON(SDL_BUTTON_LEFT)) != 0; // If a mouse press event came, always pass it as "mouse held this frame", so we don't miss click-release events that are shorter than 1 frame.
308  io.MouseDown[1] = bd->MousePressed[1] || (mouse_buttons & SDL_BUTTON(SDL_BUTTON_RIGHT)) != 0;
309  io.MouseDown[2] = bd->MousePressed[2] || (mouse_buttons & SDL_BUTTON(SDL_BUTTON_MIDDLE)) != 0;
310  bd->MousePressed[0] = bd->MousePressed[1] = bd->MousePressed[2] = false;
311 
312  // Obtain focused and hovered window. We forward mouse input when focused or when hovered (and no other window is capturing)
313 #if SDL_HAS_CAPTURE_AND_GLOBAL_MOUSE
314  SDL_Window* focused_window = SDL_GetKeyboardFocus();
315  SDL_Window* hovered_window = SDL_HAS_MOUSE_FOCUS_CLICKTHROUGH ? SDL_GetMouseFocus() : NULL; // This is better but is only reliably useful with SDL 2.0.5+ and SDL_HINT_MOUSE_FOCUS_CLICKTHROUGH.
316  SDL_Window* mouse_window = NULL;
317  if (hovered_window && bd->Window == hovered_window)
318  mouse_window = hovered_window;
319  else if (focused_window && bd->Window == focused_window)
320  mouse_window = focused_window;
321 
322  // SDL_CaptureMouse() let the OS know e.g. that our imgui drag outside the SDL window boundaries shouldn't e.g. trigger other operations outside
323  SDL_CaptureMouse(ImGui::IsAnyMouseDown() ? SDL_TRUE : SDL_FALSE);
324 #else
325  // SDL 2.0.3 and non-windowed systems: single-viewport only
326  SDL_Window* mouse_window = (SDL_GetWindowFlags(bd->Window) & SDL_WINDOW_INPUT_FOCUS) ? bd->Window : NULL;
327 #endif
328 
329  if (mouse_window == NULL)
330  return;
331 
332  // Set OS mouse position from Dear ImGui if requested (rarely used, only when ImGuiConfigFlags_NavEnableSetMousePos is enabled by user)
333  if (io.WantSetMousePos)
334  SDL_WarpMouseInWindow(bd->Window, (int)mouse_pos_prev.x, (int)mouse_pos_prev.y);
335 
336  // Set Dear ImGui mouse position from OS position + get buttons. (this is the common behavior)
337  if (bd->MouseCanUseGlobalState)
338  {
339  // Single-viewport mode: mouse position in client window coordinates (io.MousePos is (0,0) when the mouse is on the upper-left corner of the app window)
340  // Unlike local position obtained earlier this will be valid when straying out of bounds.
341  int mouse_x_global, mouse_y_global;
342  SDL_GetGlobalMouseState(&mouse_x_global, &mouse_y_global);
343  int window_x, window_y;
344  SDL_GetWindowPosition(mouse_window, &window_x, &window_y);
345  io.MousePos = ImVec2((float)(mouse_x_global - window_x), (float)(mouse_y_global - window_y));
346  }
347  else
348  {
349  io.MousePos = ImVec2((float)mouse_x_local, (float)mouse_y_local);
350  }
351 }
