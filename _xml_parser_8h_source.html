156  if (nullptr == stream) {
157  ASSIMP_LOG_DEBUG("Stream is nullptr.");
158  return false;
159  }
160 
161  const size_t len = stream->FileSize();
162  mData.resize(len + 1);
163  memset(&mData[0], '\0', len + 1);
164  stream->Read(&mData[0], 1, len);
165 
166  mDoc = new pugi::xml_document();
167  pugi::xml_parse_result parse_result = mDoc->load_string(&mData[0], pugi::parse_full);
168  if (parse_result.status == pugi::status_ok) {
169  return true;
170  }
171 
172  ASSIMP_LOG_DEBUG("Error while parse xml.", std::string(parse_result.description()), " @ ", parse_result.offset);
173 
174  return false;
175  }

375 class XmlNodeIterator {
376 public:
378  enum IterationMode {
379  PreOrderMode,
380  PostOrderMode
381  };
385  explicit XmlNodeIterator(XmlNode &parent, IterationMode mode) :
386  mParent(parent),
387  mNodes(),
388  mIndex(0) {
389  if (mode == PreOrderMode) {
390  collectChildrenPreOrder(parent);
391  } else {
392  collectChildrenPostOrder(parent);
393  }
394  }
395 
397  ~XmlNodeIterator() {
398  // empty
399  }
400 
403  void collectChildrenPreOrder( XmlNode &node ) {
404  if (node != mParent && node.type() == pugi::node_element) {
405  mNodes.push_back(node);
406  }
407  for (XmlNode currentNode : node.children()) {
408  collectChildrenPreOrder(currentNode);
409  }
410  }
411 
414  void collectChildrenPostOrder(XmlNode &node) {
415  for (XmlNode currentNode = node.first_child(); currentNode; currentNode = currentNode.next_sibling()) {
416  collectChildrenPostOrder(currentNode);
417  }
418  if (node != mParent) {
419  mNodes.push_back(node);
420  }
421  }
422 
426  bool getNext(XmlNode &next) {
427  if (mIndex == mNodes.size()) {
428  return false;
429  }
430 
431  next = mNodes[mIndex];
432  ++mIndex;
433 
434  return true;
435  }
436 
439  size_t size() const {
440  return mNodes.size();
441  }
442 
445  bool isEmpty() const {
446  return mNodes.empty();
447  }
448 
450  void clear() {
451  if (mNodes.empty()) {
452  return;
453  }
454 
455  mNodes.clear();
456  mIndex = 0;
457  }
458 
459 private:
460  XmlNode &mParent;
461  std::vector<XmlNode> mNodes;
462  size_t mIndex;
463 };
