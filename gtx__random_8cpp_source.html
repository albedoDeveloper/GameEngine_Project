70  {
71  ResultFloatA += glm::length(glm::normalizedRand3<float>());
72  ResultDoubleA += glm::length(glm::normalizedRand3<double>());
73  ResultFloatB += glm::length(glm::normalizedRand3(2.0f, 2.0f));
74  ResultDoubleB += glm::length(glm::normalizedRand3(2.0, 2.0));
75  ResultFloatC += glm::length(glm::normalizedRand3(1.0f, 3.0f));
76  ResultDoubleC += glm::length(glm::normalizedRand3(1.0, 3.0));
77  }
78 
79  Error += glm::equalEpsilon(ResultFloatA, float(Max), 0.0001f) ? 0 : 1;
80  Error += glm::equalEpsilon(ResultDoubleA, double(Max), 0.0001) ? 0 : 1;
81  Error += glm::equalEpsilon(ResultFloatB, float(Max * 2), 0.0001f) ? 0 : 1;
82  Error += glm::equalEpsilon(ResultDoubleB, double(Max * 2), 0.0001) ? 0 : 1;
83  Error += (ResultFloatC >= float(Max) && ResultFloatC <= float(Max * 3)) ? 0 : 1;
84  Error += (ResultDoubleC >= double(Max) && ResultDoubleC <= double(Max * 3)) ? 0 : 1;
85  }
