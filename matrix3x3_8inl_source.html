190 template <typename TReal>
191 AI_FORCE_INLINE
192 aiMatrix3x3t<TReal>& aiMatrix3x3t<TReal>::Inverse() {
193  // Compute the reciprocal determinant
194  TReal det = Determinant();
195  if(det == static_cast<TReal>(0.0))
196  {
197  // Matrix not invertible. Setting all elements to nan is not really
198  // correct in a mathematical sense; but at least qnans are easy to
199  // spot. XXX we might throw an exception instead, which would
200  // be even much better to spot :/.
201  const TReal nan = std::numeric_limits<TReal>::quiet_NaN();
202  *this = aiMatrix3x3t<TReal>( nan,nan,nan,nan,nan,nan,nan,nan,nan);
203 
204  return *this;
205  }
206 
207  TReal invdet = static_cast<TReal>(1.0) / det;
208 
209  aiMatrix3x3t<TReal> res;
210  res.a1 = invdet * (b2 * c3 - b3 * c2);
211  res.a2 = -invdet * (a2 * c3 - a3 * c2);
212  res.a3 = invdet * (a2 * b3 - a3 * b2);
213  res.b1 = -invdet * (b1 * c3 - b3 * c1);
214  res.b2 = invdet * (a1 * c3 - a3 * c1);
215  res.b3 = -invdet * (a1 * b3 - a3 * b1);
216  res.c1 = invdet * (b1 * c2 - b2 * c1);
217  res.c2 = -invdet * (a1 * c2 - a2 * c1);
218  res.c3 = invdet * (a1 * b2 - a2 * b1);
219  *this = res;
220 
221  return *this;
222 }

239 // Returns a rotation matrix for a rotation around an arbitrary axis.
240 template <typename TReal>
241 AI_FORCE_INLINE
242 aiMatrix3x3t<TReal>& aiMatrix3x3t<TReal>::Rotation( TReal a, const aiVector3t<TReal>& axis, aiMatrix3x3t<TReal>& out) {
243  TReal c = std::cos( a), s = std::sin( a), t = 1 - c;
244  TReal x = axis.x, y = axis.y, z = axis.z;
245 
246  // Many thanks to MathWorld and Wikipedia
247  out.a1 = t*x*x + c; out.a2 = t*x*y - s*z; out.a3 = t*x*z + s*y;
248  out.b1 = t*x*y + s*z; out.b2 = t*y*y + c; out.b3 = t*y*z - s*x;
249  out.c1 = t*x*z - s*y; out.c2 = t*y*z + s*x; out.c3 = t*z*z + c;
250 
251  return out;
252 }

274 template <typename TReal>
275 AI_FORCE_INLINE aiMatrix3x3t<TReal>& aiMatrix3x3t<TReal>::FromToMatrix(const aiVector3t<TReal>& from,
276  const aiVector3t<TReal>& to, aiMatrix3x3t<TReal>& mtx) {
277  const TReal e = from * to;
278  const TReal f = (e < 0)? -e:e;
279 
280  if (f > static_cast<TReal>(1.0) - static_cast<TReal>(0.00001)) /* "from" and "to"-vector almost parallel */
281  {
282  aiVector3D u,v; /* temporary storage vectors */
283  aiVector3D x; /* vector most nearly orthogonal to "from" */
284 
285  x.x = (from.x > 0.0)? from.x : -from.x;
286  x.y = (from.y > 0.0)? from.y : -from.y;
287  x.z = (from.z > 0.0)? from.z : -from.z;
288 
289  if (x.x < x.y)
290  {
291  if (x.x < x.z)
292  {
293  x.x = static_cast<TReal>(1.0);
294  x.y = x.z = static_cast<TReal>(0.0);
295  }
296  else
297  {
298  x.z = static_cast<TReal>(1.0);
299  x.x = x.y = static_cast<TReal>(0.0);
300  }
301  }
302  else
303  {
304  if (x.y < x.z)
305  {
306  x.y = static_cast<TReal>(1.0);
307  x.x = x.z = static_cast<TReal>(0.0);
308  }
309  else
310  {
311  x.z = static_cast<TReal>(1.0);
312  x.x = x.y = static_cast<TReal>(0.0);
313  }
314  }
315 
316  u.x = x.x - from.x; u.y = x.y - from.y; u.z = x.z - from.z;
317  v.x = x.x - to.x; v.y = x.y - to.y; v.z = x.z - to.z;
318 
319  const TReal c1_ = static_cast<TReal>(2.0) / (u * u);
320  const TReal c2_ = static_cast<TReal>(2.0) / (v * v);
321  const TReal c3_ = c1_ * c2_ * (u * v);
322 
323  for (unsigned int i = 0; i < 3; i++)
324  {
325  for (unsigned int j = 0; j < 3; j++)
326  {
327  mtx[i][j] = - c1_ * u[i] * u[j] - c2_ * v[i] * v[j]
328  + c3_ * v[i] * u[j];
329  }
330  mtx[i][i] += static_cast<TReal>(1.0);
331  }
332  }
333  else /* the most common case, unless "from"="to", or "from"=-"to" */
334  {
335  const aiVector3D v = from ^ to;
336  /* ... use this hand optimized version (9 mults less) */
337  const TReal h = static_cast<TReal>(1.0)/(static_cast<TReal>(1.0) + e); /* optimization by Gottfried Chen */
338  const TReal hvx = h * v.x;
339  const TReal hvz = h * v.z;
340  const TReal hvxy = hvx * v.y;
341  const TReal hvxz = hvx * v.z;
342  const TReal hvyz = hvz * v.y;
343  mtx[0][0] = e + hvx * v.x;
344  mtx[0][1] = hvxy - v.z;
345  mtx[0][2] = hvxz + v.y;
346 
347  mtx[1][0] = hvxy + v.z;
348  mtx[1][1] = e + h * v.y * v.y;
349  mtx[1][2] = hvyz - v.x;
350 
351  mtx[2][0] = hvxz - v.y;
352  mtx[2][1] = hvyz + v.x;
353  mtx[2][2] = e + hvz * v.z;
354  }
355  return mtx;
356 }
