53 void CRigidBody::CalcBoxInertiaTensor()
54 {
55  CCollider *col = m_parent->GetComponent<CCollider>();
56 
57  // calc inertia tensor for rectangular slab
58  m_inertiaTensor.ValuePtr()[0] = ((1 / m_inverseMass) / 12.f) * ((col->GetYHalfSize() * 2) * (col->GetYHalfSize() * 2) + (col->GetZHalfSize() * 2) * (col->GetZHalfSize() * 2)); //xx
59  m_inertiaTensor.ValuePtr()[1] = 0;
60  m_inertiaTensor.ValuePtr()[2] = 0;
61  m_inertiaTensor.ValuePtr()[3] = 0;
62  m_inertiaTensor.ValuePtr()[4] = ((1 / m_inverseMass) / 12.f) * ((col->GetXHalfSize() * 2) * (col->GetXHalfSize() * 2) + (col->GetZHalfSize() * 2) * (col->GetZHalfSize() * 2)); //yy
63  m_inertiaTensor.ValuePtr()[5] = 0;
64  m_inertiaTensor.ValuePtr()[6] = 0;
65  m_inertiaTensor.ValuePtr()[7] = 0;
66  m_inertiaTensor.ValuePtr()[8] = ((1 / m_inverseMass) / 12.f) * ((col->GetXHalfSize() * 2) * (col->GetXHalfSize() * 2) + (col->GetYHalfSize() * 2) * (col->GetYHalfSize() * 2)); //zz
67 }

69 void CRigidBody::CalcSphereInertiaTensor()
70 {
71  CCollider *col = m_parent->GetComponent<CCollider>();
72 
73  m_inertiaTensor.ValuePtr()[0] = 2.f * (1 / m_inverseMass) * col->GetXHalfSize() * col->GetXHalfSize(); //xx
74  m_inertiaTensor.ValuePtr()[1] = 0;

78  m_inertiaTensor.ValuePtr()[5] = 0;
79  m_inertiaTensor.ValuePtr()[6] = 0;
80  m_inertiaTensor.ValuePtr()[7] = 0;

216 void CRigidBody::Integrate()
217 {
218  // linear
219  m_accel = Vector3f(0, 0, 0);
220  if (m_gravityEnabled)
221  {
222  m_accel += m_gravity; // gravity
223  }
224  m_velocity += m_accel * TIME->GetDeltaTime();
225  if (m_freezeXTrans) m_velocity.SetX(0);
226  if (m_freezeYTrans) m_velocity.SetY(0);
227  if (m_freezeZTrans) m_velocity.SetZ(0);
228  m_velocity = m_velocity * powf(m_damping, TIME->GetDeltaTime()); // linear damping
229  m_parent->GetTransform()->TranslateV(m_velocity * TIME->GetDeltaTime());
230 
231  // angular
232  Matrix3f worldInverseInertiaTensor = GetInverseWorldInertiaTensor();
233  if (m_freezeXRot) m_angularVelocity.SetX(0);
234  if (m_freezeYRot) m_angularVelocity.SetY(0);
235  if (m_freezeZRot) m_angularVelocity.SetZ(0);
236  m_angularVelocity = m_angularVelocity * powf(m_damping, (TIME->GetDeltaTime())); // angular damping
237  m_parent->GetTransform()->GetRelativeOrientation().IntegrateAngVel(
238  m_angularVelocity * m_parent->GetTransform()->GetRelativeOrientation().GetInverse(),
239  TIME->GetDeltaTime()
240  );
241  m_parent->GetTransform()->GetRelativeOrientation().Normalize();
242 
243  m_parent->GetComponent<CCollider>()->UpdateCollider();
244 }
