402  std::pair<const aiTexture*, int> GetEmbeddedTextureAndIndex(const char* filename) const {
403  if(nullptr==filename) {
404  return std::make_pair(nullptr, -1);
405  }
406  // lookup using texture ID (if referenced like: "*1", "*2", etc.)
407  if ('*' == *filename) {
408  int index = std::atoi(filename + 1);
409  if (0 > index || mNumTextures <= static_cast<unsigned>(index)) {
410  return std::make_pair(nullptr, -1);
411  }
412  return std::make_pair(mTextures[index], index);
413  }
414  // lookup using filename
415  const char* shortFilename = GetShortFilename(filename);
416  if (nullptr == shortFilename) {
417  return std::make_pair(nullptr, -1);
418  }
419 
420  for (unsigned int i = 0; i < mNumTextures; i++) {
421  const char* shortTextureFilename = GetShortFilename(mTextures[i]->mFilename.C_Str());
422  if (strcmp(shortTextureFilename, shortFilename) == 0) {
423  return std::make_pair(mTextures[i], i);
424  }
425  }
426  return std::make_pair(nullptr, -1);
