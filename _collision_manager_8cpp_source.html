60 void CollisionManager::GenerateContactData()
61 {
62  m_collisionWorld.performDiscreteCollisionDetection();
63  m_contactCache.clear();
64 
65  for (unsigned i = 0; i < m_collisionWorld.getDispatcher()->getNumManifolds(); i++)
66  {
67  if (m_collisionWorld.getDispatcher()->getManifoldByIndexInternal(i)->getNumContacts() < 1) // leave loop if manifold has no contacts
68  {
69  continue;
70  }
71  if (!GetCColliderRegistration(m_collisionWorld.getDispatcher()->getManifoldByIndexInternal(i)->getBody0())->GetParentObject()->GetCRigidBody() &&
72  !GetCColliderRegistration(m_collisionWorld.getDispatcher()->getManifoldByIndexInternal(i)->getBody1())->GetParentObject()->GetCRigidBody() // don't generate contacts for static scenery collisions
73  )
74  {
75  continue;
76  }
77 
78  // ensure body 1 always has a rigidbody
79  if (GetCColliderRegistration(m_collisionWorld.getDispatcher()->getManifoldByIndexInternal(i)->getBody0())->GetParentObject()->GetCRigidBody())
80  {
81  FillManifoldAB(i);
82  }
83  else
84  {
85  FillManifoldBA(i);
86  }
87  assert(m_contactCache.back().GetCol1()->GetParentObject()->GetCRigidBody());
88 
89  }
