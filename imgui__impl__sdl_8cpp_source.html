111 // You can read the io.WantCaptureMouse, io.WantCaptureKeyboard flags to tell if dear imgui wants to use your inputs.
112 // - When io.WantCaptureMouse is true, do not dispatch mouse input data to your main application.
113 // - When io.WantCaptureKeyboard is true, do not dispatch keyboard input data to your main application.
114 // Generally you may always pass all inputs to dear imgui, and hide them from your application based on those two flags.
115 // If you have multiple SDL events and some of them are not meant to be used by dear imgui, you may need to filter events based on their windowID field.
116 bool ImGui_ImplSDL2_ProcessEvent(const SDL_Event *event)
117 {
118  ImGuiIO &io = ImGui::GetIO();
119  ImGui_ImplSDL2_Data *bd = ImGui_ImplSDL2_GetBackendData();
120 
121  switch (event->type)
122  {
123  case SDL_MOUSEWHEEL:
124  {
125  if (event->wheel.x > 0) io.MouseWheelH += 1;
126  if (event->wheel.x < 0) io.MouseWheelH -= 1;
127  if (event->wheel.y > 0) io.MouseWheel += 1;
128  if (event->wheel.y < 0) io.MouseWheel -= 1;
129  return true;
130  }
131  case SDL_MOUSEBUTTONDOWN:
132  {
133  if (event->button.button == SDL_BUTTON_LEFT)
134  {
135  bd->MousePressed[0] = true;
136  }
137  if (event->button.button == SDL_BUTTON_RIGHT)
138  {
139  bd->MousePressed[1] = true;
140  }
141  if (event->button.button == SDL_BUTTON_MIDDLE)
142  {
143  bd->MousePressed[2] = true;
144  }
145  return true;
146  }
147  case SDL_TEXTINPUT:
148  {
149  io.AddInputCharactersUTF8(event->text.text);
150  return true;
151  }
152  case SDL_KEYDOWN:
153  case SDL_KEYUP:
154  {
155  int key = event->key.keysym.scancode;
156  IM_ASSERT(key >= 0 && key < IM_ARRAYSIZE(io.KeysDown));
157  io.KeysDown[key] = (event->type == SDL_KEYDOWN);
158  io.KeyShift = ((SDL_GetModState() & KMOD_SHIFT) != 0);
159  io.KeyCtrl = ((SDL_GetModState() & KMOD_CTRL) != 0);
160  io.KeyAlt = ((SDL_GetModState() & KMOD_ALT) != 0);
161 #ifdef _WIN32
162  io.KeySuper = false;
163 #else
164  io.KeySuper = ((SDL_GetModState() & KMOD_GUI) != 0);
165 #endif
166  return true;
167  }
168  case SDL_WINDOWEVENT:
169  {
170  if (event->window.event == SDL_WINDOWEVENT_FOCUS_GAINED)
171  io.AddFocusEvent(true);
172  else if (event->window.event == SDL_WINDOWEVENT_FOCUS_LOST)
173  io.AddFocusEvent(false);
174  return true;
175  }
176  }
177  return false;
178 }

308 static void ImGui_ImplSDL2_UpdateMousePosAndButtons()
309 {
310  ImGui_ImplSDL2_Data *bd = ImGui_ImplSDL2_GetBackendData();
311  ImGuiIO &io = ImGui::GetIO();
312 
313  ImVec2 mouse_pos_prev = io.MousePos;
314  io.MousePos = ImVec2(-FLT_MAX, -FLT_MAX);
315 
316  // Update mouse buttons
317  int mouse_x_local, mouse_y_local;
318  Uint32 mouse_buttons = SDL_GetMouseState(&mouse_x_local, &mouse_y_local);
319  io.MouseDown[0] = bd->MousePressed[0] || (mouse_buttons & SDL_BUTTON(SDL_BUTTON_LEFT)) != 0; // If a mouse press event came, always pass it as "mouse held this frame", so we don't miss click-release events that are shorter than 1 frame.
320  io.MouseDown[1] = bd->MousePressed[1] || (mouse_buttons & SDL_BUTTON(SDL_BUTTON_RIGHT)) != 0;
321  io.MouseDown[2] = bd->MousePressed[2] || (mouse_buttons & SDL_BUTTON(SDL_BUTTON_MIDDLE)) != 0;
322  bd->MousePressed[0] = bd->MousePressed[1] = bd->MousePressed[2] = false;
323 
324  // Obtain focused and hovered window. We forward mouse input when focused or when hovered (and no other window is capturing)
325 #if SDL_HAS_CAPTURE_AND_GLOBAL_MOUSE
326  SDL_Window *focused_window = SDL_GetKeyboardFocus();
327  SDL_Window *hovered_window = SDL_HAS_MOUSE_FOCUS_CLICKTHROUGH ? SDL_GetMouseFocus() : NULL; // This is better but is only reliably useful with SDL 2.0.5+ and SDL_HINT_MOUSE_FOCUS_CLICKTHROUGH.
328  SDL_Window *mouse_window = NULL;
329  if (hovered_window && bd->Window == hovered_window)
330  mouse_window = hovered_window;
331  else if (focused_window && bd->Window == focused_window)
332  mouse_window = focused_window;
333 
334  // SDL_CaptureMouse() let the OS know e.g. that our imgui drag outside the SDL window boundaries shouldn't e.g. trigger other operations outside
335  SDL_CaptureMouse(ImGui::IsAnyMouseDown() ? SDL_TRUE : SDL_FALSE);
336 #else
337  // SDL 2.0.3 and non-windowed systems: single-viewport only
338  SDL_Window *mouse_window = (SDL_GetWindowFlags(bd->Window) & SDL_WINDOW_INPUT_FOCUS) ? bd->Window : NULL;
339 #endif
340 
341  if (mouse_window == NULL)
342  return;
343 
344  // Set OS mouse position from Dear ImGui if requested (rarely used, only when ImGuiConfigFlags_NavEnableSetMousePos is enabled by user)
345  if (io.WantSetMousePos)
346  SDL_WarpMouseInWindow(bd->Window, (int)mouse_pos_prev.x, (int)mouse_pos_prev.y);
347 
348  // Set Dear ImGui mouse position from OS position + get buttons. (this is the common behavior)
349  if (bd->MouseCanUseGlobalState)
350  {
351  // Single-viewport mode: mouse position in client window coordinates (io.MousePos is (0,0) when the mouse is on the upper-left corner of the app window)
352  // Unlike local position obtained earlier this will be valid when straying out of bounds.
353  int mouse_x_global, mouse_y_global;
354  SDL_GetGlobalMouseState(&mouse_x_global, &mouse_y_global);
355  int window_x, window_y;
356  SDL_GetWindowPosition(mouse_window, &window_x, &window_y);
357  io.MousePos = ImVec2((float)(mouse_x_global - window_x), (float)(mouse_y_global - window_y));
358  }
359  else
360  {
361  io.MousePos = ImVec2((float)mouse_x_local, (float)mouse_y_local);
362  }
363 }

434  int w, h;
435  int display_w, display_h;
436  SDL_GetWindowSize(bd->Window, &w, &h);
437  if (SDL_GetWindowFlags(bd->Window) & SDL_WINDOW_MINIMIZED)
438  w = h = 0;

441  if (w > 0 && h > 0)
442  io.DisplayFramebufferScale = ImVec2((float)display_w / w, (float)display_h / h);
443 
444  // Setup time step (we don't use SDL_GetTicks() because it is using millisecond resolution)
445  static Uint64 frequency = SDL_GetPerformanceFrequency();
446  Uint64 current_time = SDL_GetPerformanceCounter();
447  io.Time = bd->Time > 0 ? (float)((double)(current_time - bd->Time) / frequency) : (float)(1.0f / 60.0f);
448  bd->Time = current_time;
449

451  ImGui_ImplSDL2_UpdateMouseCursor();
