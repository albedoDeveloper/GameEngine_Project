81 inline unsigned int ASSIMP_itoa10(char *out, unsigned int max, int32_t number) {
82  ai_assert(nullptr != out);
83 
84  // write the unary minus to indicate we have a negative number
85  unsigned int written = 1u;
86  if (number < 0 && written < max) {
87  *out++ = '-';
88  ++written;
89  number = -number;
90  }
91 
92  // We begin with the largest number that is not zero.
93  int32_t cur = 1000000000; // 2147483648
94  bool mustPrint = false;
95  while (written < max) {
96 
97  const unsigned int digit = number / cur;
98  if (mustPrint || digit > 0 || 1 == cur) {
99  // print all future zero's from now
100  mustPrint = true;
101 
102  *out++ = '0' + static_cast<char>(digit);
103 
104  ++written;
105  number -= digit * cur;
106  if (1 == cur) {
107  break;
108  }
109  }
110  cur /= 10;
111  }
112 
113  // append a terminal zero
114  *out++ = '\0';
115  return written - 1;
116 }

139 inline int ASSIMP_stricmp(const char *s1, const char *s2) {
140  ai_assert(nullptr != s1);
141  ai_assert(nullptr != s2);
142 
143 #if (defined _MSC_VER)
144 
145  return ::_stricmp(s1, s2);
146 #else
147  char c1, c2;
148  do {
149  c1 = tolower((unsigned char)*(s1++));
150  c2 = tolower((unsigned char)*(s2++));
151  } while (c1 && (c1 == c2));
152  return c1 - c2;
153 #endif
154 }

180 inline int ASSIMP_strincmp(const char *s1, const char *s2, unsigned int n) {
181  ai_assert(nullptr != s1);
182  ai_assert(nullptr != s2);
183  if (!n) {
184  return 0;
185  }
186 
187 #if (defined _MSC_VER)
188 
189  return ::_strnicmp(s1, s2, n);
190 
191 #elif defined(__GNUC__)
192 
193  return ::strncasecmp(s1, s2, n);
194 
195 #else
196  char c1, c2;
197  unsigned int p = 0;
198  do {
199  if (p++ >= n) return 0;
200  c1 = tolower((unsigned char)*(s1++));
201  c2 = tolower((unsigned char)*(s2++));
202  } while (c1 && (c1 == c2));
203 
204  return c1 - c2;
205 #endif
206 }
