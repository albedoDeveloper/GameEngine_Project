330 template<>
331 struct Stack<std::string>
332 {
333  static void push(lua_State* L, std::string const& str)
334  {
335  lua_pushlstring(L, str.data(), str.size());
336  }
337 
338  static std::string get(lua_State* L, int index)
339  {
340  size_t len;
341  if (lua_type(L, index) == LUA_TSTRING)
342  {
343  const char* str = lua_tolstring(L, index, &len);
344  return std::string(str, len);
345  }
346 
347  // Lua reference manual:
348  // If the value is a number, then lua_tolstring also changes the actual value in the stack
349  // to a string. (This change confuses lua_next when lua_tolstring is applied to keys during
350  // a table traversal.)
351  lua_pushvalue(L, index);
352  const char* str = lua_tolstring(L, -1, &len);
353  std::string string(str, len);
354  lua_pop(L, 1); // Pop the temporary string
355  return string;
356  }
357 
358  static bool isInstance(lua_State* L, int index) { return lua_type(L, index) == LUA_TSTRING; }
359 };
