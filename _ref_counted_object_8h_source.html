74 template<class CounterType>
75 class RefCountedObjectType
76 {
77 public:
78  //==============================================================================
84  inline void incReferenceCount() const { ++refCount; }
85 
90  inline void decReferenceCount() const
91  {
92  assert(getReferenceCount() > 0);
93 
94  if (--refCount == 0)
95  delete this;
96  }
97 
101  inline int getReferenceCount() const { return static_cast<int>(refCount); }
102 
103 protected:
104  //==============================================================================
106  RefCountedObjectType() : refCount() {}
107 
109  virtual ~RefCountedObjectType()
110  {
111  // it's dangerous to delete an object that's still referenced by something else!
112  assert(getReferenceCount() == 0);
113  }
114 
115 private:
116  //==============================================================================
117  CounterType mutable refCount;
118 };

252  RefCountedObjectPtr& operator=(ReferenceCountedObjectClass* const newObject)
