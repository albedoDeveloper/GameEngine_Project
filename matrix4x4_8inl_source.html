227 AI_FORCE_INLINE
228 TReal aiMatrix4x4t<TReal>::Determinant() const {
229  return a1*b2*c3*d4 - a1*b2*c4*d3 + a1*b3*c4*d2 - a1*b3*c2*d4
230  + a1*b4*c2*d3 - a1*b4*c3*d2 - a2*b3*c4*d1 + a2*b3*c1*d4
231  - a2*b4*c1*d3 + a2*b4*c3*d1 - a2*b1*c3*d4 + a2*b1*c4*d3
232  + a3*b4*c1*d2 - a3*b4*c2*d1 + a3*b1*c2*d4 - a3*b1*c4*d2
233  + a3*b2*c4*d1 - a3*b2*c1*d4 - a4*b1*c2*d3 + a4*b1*c3*d2
234  - a4*b2*c3*d1 + a4*b2*c1*d3 - a4*b3*c1*d2 + a4*b3*c2*d1;

238 template <typename TReal>
239 AI_FORCE_INLINE
240 aiMatrix4x4t<TReal>& aiMatrix4x4t<TReal>::Inverse() {
241  // Compute the reciprocal determinant
242  const TReal det = Determinant();
243  if(det == static_cast<TReal>(0.0))
244  {
245  // Matrix not invertible. Setting all elements to nan is not really
246  // correct in a mathematical sense but it is easy to debug for the
247  // programmer.
248  const TReal nan = std::numeric_limits<TReal>::quiet_NaN();
249  *this = aiMatrix4x4t<TReal>(
250  nan,nan,nan,nan,
251  nan,nan,nan,nan,
252  nan,nan,nan,nan,
253  nan,nan,nan,nan);
254 
255  return *this;
256  }
257 
258  const TReal invdet = static_cast<TReal>(1.0) / det;
259 
260  aiMatrix4x4t<TReal> res;
261  res.a1 = invdet * (b2 * (c3 * d4 - c4 * d3) + b3 * (c4 * d2 - c2 * d4) + b4 * (c2 * d3 - c3 * d2));
262  res.a2 = -invdet * (a2 * (c3 * d4 - c4 * d3) + a3 * (c4 * d2 - c2 * d4) + a4 * (c2 * d3 - c3 * d2));
263  res.a3 = invdet * (a2 * (b3 * d4 - b4 * d3) + a3 * (b4 * d2 - b2 * d4) + a4 * (b2 * d3 - b3 * d2));
264  res.a4 = -invdet * (a2 * (b3 * c4 - b4 * c3) + a3 * (b4 * c2 - b2 * c4) + a4 * (b2 * c3 - b3 * c2));
265  res.b1 = -invdet * (b1 * (c3 * d4 - c4 * d3) + b3 * (c4 * d1 - c1 * d4) + b4 * (c1 * d3 - c3 * d1));
266  res.b2 = invdet * (a1 * (c3 * d4 - c4 * d3) + a3 * (c4 * d1 - c1 * d4) + a4 * (c1 * d3 - c3 * d1));
267  res.b3 = -invdet * (a1 * (b3 * d4 - b4 * d3) + a3 * (b4 * d1 - b1 * d4) + a4 * (b1 * d3 - b3 * d1));
268  res.b4 = invdet * (a1 * (b3 * c4 - b4 * c3) + a3 * (b4 * c1 - b1 * c4) + a4 * (b1 * c3 - b3 * c1));
269  res.c1 = invdet * (b1 * (c2 * d4 - c4 * d2) + b2 * (c4 * d1 - c1 * d4) + b4 * (c1 * d2 - c2 * d1));
270  res.c2 = -invdet * (a1 * (c2 * d4 - c4 * d2) + a2 * (c4 * d1 - c1 * d4) + a4 * (c1 * d2 - c2 * d1));
271  res.c3 = invdet * (a1 * (b2 * d4 - b4 * d2) + a2 * (b4 * d1 - b1 * d4) + a4 * (b1 * d2 - b2 * d1));
272  res.c4 = -invdet * (a1 * (b2 * c4 - b4 * c2) + a2 * (b4 * c1 - b1 * c4) + a4 * (b1 * c2 - b2 * c1));
273  res.d1 = -invdet * (b1 * (c2 * d3 - c3 * d2) + b2 * (c3 * d1 - c1 * d3) + b3 * (c1 * d2 - c2 * d1));
274  res.d2 = invdet * (a1 * (c2 * d3 - c3 * d2) + a2 * (c3 * d1 - c1 * d3) + a3 * (c1 * d2 - c2 * d1));
275  res.d3 = -invdet * (a1 * (b2 * d3 - b3 * d2) + a2 * (b3 * d1 - b1 * d3) + a3 * (b1 * d2 - b2 * d1));
276  res.d4 = invdet * (a1 * (b2 * c3 - b3 * c2) + a2 * (b3 * c1 - b1 * c3) + a3 * (b1 * c2 - b2 * c1));
277  *this = res;
278 
279  return *this;
280 }

370  const aiMatrix4x4t<TReal>& _this = *this;/* Create alias for conveniance. */ \
371  \
372  /* extract translation */ \
373  pPosition.x = _this[0][3]; \
374  pPosition.y = _this[1][3]; \
375  pPosition.z = _this[2][3]; \
376  \
377  /* extract the columns of the matrix. */ \
378  aiVector3t<TReal> vCols[3] = { \
379  aiVector3t<TReal>(_this[0][0],_this[1][0],_this[2][0]), \
380  aiVector3t<TReal>(_this[0][1],_this[1][1],_this[2][1]), \
381  aiVector3t<TReal>(_this[0][2],_this[1][2],_this[2][2]) \
382  }; \
383  \
384  /* extract the scaling factors */ \
385  pScaling.x = vCols[0].Length(); \
386  pScaling.y = vCols[1].Length(); \
387  pScaling.z = vCols[2].Length(); \
388  \
389  /* and the sign of the scaling */ \
390  if (Determinant() < 0) pScaling = -pScaling; \
391  \
392  /* and remove all scaling from the matrix */ \
393  if(pScaling.x) vCols[0] /= pScaling.x; \
394  if(pScaling.y) vCols[1] /= pScaling.y; \
395  if(pScaling.z) vCols[2] /= pScaling.z; \
396  \
397  do {} while(false)

414 template <typename TReal>
415 AI_FORCE_INLINE
416 void aiMatrix4x4t<TReal>::Decompose(aiVector3t<TReal>& pScaling, aiVector3t<TReal>& pRotation, aiVector3t<TReal>& pPosition) const {
417  ASSIMP_MATRIX4_4_DECOMPOSE_PART;
418 
419  /*
420  assuming a right-handed coordinate system
421  and post-multiplication of column vectors,
422  the rotation matrix for an euler XYZ rotation is M = Rz * Ry * Rx.
423  combining gives:
424 
425  | CE BDE-AF ADE+BF 0 |
426  M = | CF BDF+AE ADF-BE 0 |
427  | -D CB AC 0 |
428  | 0 0 0 1 |
429 
430  where
431  A = cos(angle_x), B = sin(angle_x);
432  C = cos(angle_y), D = sin(angle_y);
433  E = cos(angle_z), F = sin(angle_z);
434  */
435 
436  // Use a small epsilon to solve floating-point inaccuracies
437  const TReal epsilon = Assimp::Math::getEpsilon<TReal>();
438 
439  pRotation.y = std::asin(-vCols[0].z);// D. Angle around oY.
440 
441  TReal C = std::cos(pRotation.y);
442 
443  if(std::fabs(C) > epsilon)
444  {
445  // Finding angle around oX.
446  TReal tan_x = vCols[2].z / C;// A
447  TReal tan_y = vCols[1].z / C;// B
448 
449  pRotation.x = std::atan2(tan_y, tan_x);
450  // Finding angle around oZ.
451  tan_x = vCols[0].x / C;// E
452  tan_y = vCols[0].y / C;// F
453  pRotation.z = std::atan2(tan_y, tan_x);
454  }
455  else
456  {// oY is fixed.
457  pRotation.x = 0;// Set angle around oX to 0. => A == 1, B == 0, C == 0, D == 1.
458 
459  // And finding angle around oZ.
460  TReal tan_x = vCols[1].y;// BDF+AE => E
461  TReal tan_y = -vCols[1].x;// BDE-AF => F
462 
463  pRotation.z = std::atan2(tan_y, tan_x);
464  }
465 }

467 #undef ASSIMP_MATRIX4_4_DECOMPOSE_PART

469 template <typename TReal>
470 AI_FORCE_INLINE
471 void aiMatrix4x4t<TReal>::Decompose(aiVector3t<TReal>& pScaling, aiVector3t<TReal>& pRotationAxis, TReal& pRotationAngle,
472  aiVector3t<TReal>& pPosition) const {
473  aiQuaterniont<TReal> pRotation;
474 
475  Decompose(pScaling, pRotation, pPosition);
476  pRotation.Normalize();
477 
478  TReal angle_cos = pRotation.w;
479  TReal angle_sin = std::sqrt(1.0f - angle_cos * angle_cos);
480 
481  pRotationAngle = std::acos(angle_cos) * 2;
482 
483  // Use a small epsilon to solve floating-point inaccuracies
484  const TReal epsilon = 10e-3f;
485 
486  if(std::fabs(angle_sin) < epsilon) angle_sin = 1;
487 
488  pRotationAxis.x = pRotation.x / angle_sin;
489  pRotationAxis.y = pRotation.y / angle_sin;
490  pRotationAxis.z = pRotation.z / angle_sin;
491 }
